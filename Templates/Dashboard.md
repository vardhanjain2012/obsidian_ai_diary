```dataviewjs
dv.span("** 🗓️ Daily Ingestion Tracker **");

// 1. Initialize Heatmap Settings
const currentYear = moment().year();
const calendarData = {
    year: currentYear,
    entries: [],
    colors: {
        green: ["#4caf50"],  
        yellow: ["#ffc107"], 
        red: ["#f44336"],    
    },
    showCurrentTrackingContext: true
};

// 2. Fetch all daily notes from the specific folder (Optimized)
const pages = dv.pages('"daily_notes"'); 
const existingDates = new Set();

// 3. Process existing files
for (let page of pages) {
    if (!moment(page.file.name, "YYYY-MM-DD", true).isValid()) continue;

    let dateStr = page.file.name;
    existingDates.add(dateStr);
    
    let content = await dv.io.load(page.file.path);

    let brainDumpSplit = content.split("## 🧠 Brain Dump");
    let dumpText = brainDumpSplit.length > 1 ? brainDumpSplit[1].trim() : "";
    let hasDump = dumpText.length > 15; 

    // Look for the word "Skipped" as generated by your JS template
    let questionsSkipped = content.includes("Skipped");

    let entryColor = "red"; 
    
    if (hasDump && !questionsSkipped) {
        entryColor = "green";
    } else if (hasDump && questionsSkipped) {
        entryColor = "yellow";
    }

    calendarData.entries.push({
        date: dateStr,
        intensity: 1, 
        color: entryColor,
        // Uses the dynamic exact path from the file object
        content: await dv.span(`[[${page.file.path}|&nbsp;]]`) 
    });
}

// 4. Backfill missing days (Red if file doesn't exist)
const startOfYear = moment().startOf('year');
const today = moment();

for (let m = moment(startOfYear); m.isSameOrBefore(today); m.add(1, 'days')) {
    let dateStr = m.format("YYYY-MM-DD");
    
    if (!existingDates.has(dateStr)) {
        calendarData.entries.push({
            date: dateStr,
            intensity: 1,
            color: "red",
            // Hardcoded to the correct new directory structure
            content: await dv.span(`[[daily_notes/${dateStr}|&nbsp;]]`) 
        });
    }
}

// 5. Render the final Heatmap
renderHeatmapCalendar(this.container, calendarData);
```
